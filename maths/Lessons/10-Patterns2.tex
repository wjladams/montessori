\section*{Exercises}

\begin{enumerate}
	\item $1,6,11,16,\ldots$
	\item $1, 4, 9, 16, \ldots$
	\item $1, 2, 3, 5, 7, 10, 13, 17, 21, 26, \ldots$

	\item $3, 6, 12, 24, 48,\ldots$

	\item $12, 21, 30, 39, \ldots$
